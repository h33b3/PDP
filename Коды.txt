def login_view(request):
    if request.method == 'POST':
        login_input = request.POST.get('login')
        password_input = request.POST.get('password')

        try:
            user = UserAccount.objects.get(login=login_input)
            if check_password(password_input, user.password):
                request.session['user_id'] = user.user_id
                request.session['user_login'] = user.login
                request.session['user_role'] = user.role.name if user.role else 'user'
                return redirect('appeal_list')
            else:
                messages.error(request, 'Неверный логин или пароль')
        except UserAccount.DoesNotExist:
            messages.error(request, 'Неверный логин или пароль')

    return render(request, 'appeals/login.html')


def logout_view(request):
    request.session.flush()
    return redirect('login')


def appeal_list(request):
    if not request.session.get('user_id'):
        return redirect('login')

    status_filter = request.GET.get('status')
    search_query = request.GET.get('search', '')
    sort_by = request.GET.get('sort', 'newest')

    appeals = Appeal.objects.all().select_related(
        'status', 'department', 'method', 'executor', 'address'
    )

    if status_filter and status_filter != 'all':
        appeals = appeals.filter(status__name=status_filter)

    if search_query:
        appeals = appeals.filter(
            Q(appeal_id__icontains=search_query) |
            Q(topic__icontains=search_query) |
            Q(description__icontains=search_query) |
            Q(applicant_last_name__icontains=search_query) |
            Q(applicant_first_name__icontains=search_query) |
            Q(applicant_middle_name__icontains=search_query) |
            Q(phone__icontains=search_query) |
            Q(email__icontains=search_query) |
            Q(department__name__icontains=search_query) |
            Q(status__name__icontains=search_query) |
            Q(executor__last_name__icontains=search_query) |
            Q(executor__first_name__icontains=search_query) |
            Q(method__name__icontains=search_query) |
            Q(address__country__icontains=search_query) |
            Q(address__city__icontains=search_query) |
            Q(address__street__icontains=search_query)
        )

    if sort_by == 'oldest':
        appeals = appeals.order_by('created_date')
    else:
        appeals = appeals.order_by('-created_date')

    context = {
        'appeals': appeals,
        'search_query': search_query,
        'current_filter': status_filter if status_filter else 'all',
        'date_filter': sort_by,
        'active_tab': 'appeals',
    }
    return render(request, 'appeals/appeal_list.html', context)


def search_ajax(request):
    if not request.session.get('user_id'):
        return JsonResponse({'success': False, 'error': 'Unauthorized'}, status=401)
    
    query = request.GET.get('q', '').strip()
    
    if len(query) < 2:
        return JsonResponse({'success': True, 'results': []})
    
    results = Appeal.objects.filter(
        Q(appeal_id__icontains=query) |
        Q(topic__icontains=query) |
        Q(applicant_last_name__icontains=query) |
        Q(applicant_first_name__icontains=query) |
        Q(phone__icontains=query) |
        Q(email__icontains=query) |
        Q(department__name__icontains=query) |
        Q(status__name__icontains=query)
    ).select_related('status', 'department')[:10]
    
    results_data = []
    for appeal in results:
        results_data.append({
            'id': appeal.appeal_id,
            'appeal_id': appeal.appeal_id,
            'topic': appeal.topic if appeal.topic else 'Без темы',
            'status': appeal.status.name if appeal.status else 'Неизвестно',
            'department': appeal.department.name if appeal.department else 'Не назначен',
            'applicant': f"{appeal.applicant_last_name} {appeal.applicant_first_name}".strip(),
        })
    
    return JsonResponse({'success': True, 'results': results_data})

def appeal_create(request):
    if not request.session.get('user_id'):
        return redirect('login')

    if request.method == 'POST':
        appeal_form = AppealForm(request.POST)

        if appeal_form.is_valid():
            try:
                address = Address.objects.create(
                    country=request.POST.get('country', ''),
                    city=request.POST.get('city', ''),
                    street=request.POST.get('street', ''),
                    building=request.POST.get('building', ''),
                    apartment=request.POST.get('apartment', '')
                )
                
                appeal = appeal_form.save(commit=False)
                appeal.address = address
                appeal.operator_id = request.session.get('user_id')
                appeal.created_date = timezone.now().date()
                appeal.save()

                uploaded_files = request.FILES.getlist('attachments')
                for uploaded_file in uploaded_files:
                    upload_dir = os.path.join(settings.MEDIA_ROOT, 'attachments', str(appeal.appeal_id))
                    os.makedirs(upload_dir, exist_ok=True)
                    
                    file_path = os.path.join(upload_dir, uploaded_file.name)
                    with open(file_path, 'wb+') as destination:
                        for chunk in uploaded_file.chunks():
                            destination.write(chunk)
                    
                    attachment = Attachment.objects.create(
                        file_name=uploaded_file.name,
                        file_path=file_path,
                        upload_date=timezone.now().date()
                    )
                    appeal.attachment = attachment
                    appeal.save()
                
                History.objects.create(
                    appeal=appeal,
                    action='Обращение зарегистрировано',
                    comment=f'Оператор: {request.session["user_login"]}'
                )
                
                messages.success(request, f'Обращение №{appeal.appeal_id} успешно создано!')
                return redirect('appeal_detail', pk=appeal.appeal_id)
                
            except Exception as e:
                messages.error(request, f'Ошибка: {str(e)}')
        else:
            messages.error(request, 'Исправьте ошибки в форме')
    
    context = {
        'appeal_form': AppealForm(),
        'methods': Method.objects.all(),
        'statuses': Status.objects.all(),
        'departments': Department.objects.all(),
        'executors': Executor.objects.all(),
        'active_tab': 'appeals',
    }
    return render(request, 'appeals/appeal_create.html', context)


def appeal_edit(request, pk):
    if not request.session.get('user_id'):
        return redirect('login')

    appeal = get_object_or_404(Appeal, pk=pk)
    
    if request.method == 'POST':
        form = AppealForm(request.POST, instance=appeal)
        
        if form.is_valid():
            # Обновляем адрес
            if appeal.address:
                address = appeal.address
                address.country = request.POST.get('country', '')
                address.city = request.POST.get('city', '')
                address.street = request.POST.get('street', '')
                address.building = request.POST.get('building', '')
                address.apartment = request.POST.get('apartment', '')
                address.save()
            else:
                address = Address.objects.create(
                    country=request.POST.get('country', ''),
                    city=request.POST.get('city', ''),
                    street=request.POST.get('street', ''),
                    building=request.POST.get('building', ''),
                    apartment=request.POST.get('apartment', '')
                )
            
            appeal = form.save(commit=False)
            appeal.address = address
            appeal.save()
            
            # Обработка файлов
            uploaded_files = request.FILES.getlist('attachments')
            for uploaded_file in uploaded_files:
                upload_dir = os.path.join(settings.MEDIA_ROOT, 'attachments', str(appeal.appeal_id))
                os.makedirs(upload_dir, exist_ok=True)
                
                file_path = os.path.join(upload_dir, uploaded_file.name)
                with open(file_path, 'wb+') as destination:
                    for chunk in uploaded_file.chunks():
                        destination.write(chunk)
                
                attachment = Attachment.objects.create(
                    file_name=uploaded_file.name,
                    file_path=file_path,
                    upload_date=timezone.now().date()
                )
                appeal.attachment = attachment
                appeal.save()
            
            History.objects.create(
                appeal=appeal,
                action='Обращение изменено',
                comment=f'Изменения внес: {request.session["user_login"]}'
            )
            messages.success(request, 'Обращение обновлено')
            return redirect('appeal_detail', pk=appeal.appeal_id)
    else:
        form = AppealForm(instance=appeal)

    context = {
        'appeal_form': form,
        'appeal': appeal,
        'methods': Method.objects.all(),
        'statuses': Status.objects.all(),
        'departments': Department.objects.all(),
        'executors': Executor.objects.all(),
        'active_tab': 'appeals',
    }
    return render(request, 'appeals/appeal_create.html', context)


def add_response(request, pk):
    if not request.session.get('user_id'):
        return redirect('login')

    appeal = get_object_or_404(Appeal, pk=pk)
    
    if request.method == 'POST':
        response_text = request.POST.get('response_text')
        send_status = request.POST.get('send_status', 'not sent')

        response, created = Response.objects.update_or_create(
            appeal=appeal,
            defaults={
                'response_text': response_text,
                'response_status': send_status,
                'response_date': timezone.now().date()
            }
        )

        if send_status == 'sent':
            try:
                completed_status = Status.objects.get(name='Завершенное')
                appeal.status = completed_status
                appeal.save()
            except Status.DoesNotExist:
                pass

        History.objects.create(
            appeal=appeal,
            action='Добавлен ответ',
            comment=f'Ответ {"отправлен" if send_status == "sent" else "сохранен как черновик"}: {response_text[:100]}'
        )
        
        if send_status == 'sent':
            messages.success(request, 'Ответ успешно отправлен заявителю!')
        else:
            messages.success(request, 'Ответ сохранен как черновик. Вы можете отредактировать его позже.')
        
        return redirect('appeal_detail', pk=appeal.appeal_id)
    
    context = {
        'appeal': appeal,
        'active_tab': 'appeals',
    }
    return render(request, 'appeals/add_response.html', context)

def generate_report(request):
    if not request.session.get('user_id'):
        return redirect('login')
    
    if request.method != 'POST':
        return redirect('report')
    
    period_start = request.POST.get('period_start')
    period_end = request.POST.get('period_end')
    department_id = request.POST.get('department')
    status_id = request.POST.get('status')
    method_id = request.POST.get('method')
    executor_id = request.POST.get('executor')
    report_type = request.POST.get('report_type', 'detailed')
    report_format = request.POST.get('format', 'excel')
    
    appeals = Appeal.objects.all().select_related('status', 'department', 'method', 'executor', 'address')
    
    if period_start:
        appeals = appeals.filter(created_date__gte=period_start)
    if period_end:
        appeals = appeals.filter(created_date__lte=period_end)
    if department_id:
        appeals = appeals.filter(department_id=department_id)
    if status_id:
        appeals = appeals.filter(status_id=status_id)
    if method_id:
        appeals = appeals.filter(method_id=method_id)
    if executor_id:
        appeals = appeals.filter(executor_id=executor_id)
    
    if report_format == 'excel':
        return generate_excel_report(appeals, report_type, request)
    else:
        return generate_pdf_report(appeals, report_type, request)


def generate_excel_report(appeals, report_type, request):
    try:
        import openpyxl
        from openpyxl.styles import Font, Alignment, PatternFill, Border, Side
    except ImportError:
        return HttpResponse('Библиотека openpyxl не установлена. Выполните: pip install openpyxl', status=500)
    
    wb = openpyxl.Workbook()
    wb.remove(wb.active)
    
    header_font = Font(bold=True, size=12, color="FFFFFF")
    header_fill = PatternFill(start_color="1a3a5c", end_color="1a3a5c", fill_type="solid")
    header_alignment = Alignment(horizontal="center", vertical="center")
    border = Border(
        left=Side(style='thin'),
        right=Side(style='thin'),
        top=Side(style='thin'),
        bottom=Side(style='thin')
    )
    
    if report_type == 'detailed':
        ws = wb.create_sheet("Детальный отчет")
        
        headers = ['№', 'Номер', 'Дата создания', 'Тема', 'Заявитель', 'Телефон', 'Email', 
                   'Отдел', 'Статус', 'Тип обращения', 'Исполнитель', 'Срок исполнения', 'Адрес']
        for col_idx, header in enumerate(headers, start=1):
            cell = ws.cell(row=1, column=col_idx, value=header)
            cell.font = header_font
            cell.fill = header_fill
            cell.alignment = header_alignment
            cell.border = border
        
        for row_idx, appeal in enumerate(appeals, start=2):
            data = [
                row_idx - 1,
                appeal.appeal_id,
                appeal.created_date.strftime('%d.%m.%Y') if appeal.created_date else '-',
                appeal.topic,
                f"{appeal.applicant_last_name} {appeal.applicant_first_name}",
                appeal.phone or '-',
                appeal.email or '-',
                appeal.department.name if appeal.department else '-',
                appeal.status.name if appeal.status else '-',
                appeal.method.name if appeal.method else '-',
                str(appeal.executor) if appeal.executor else '-',
                appeal.deadline.strftime('%d.%m.%Y') if appeal.deadline else '-',
                str(appeal.address) if appeal.address else '-',
            ]
            for col_idx, value in enumerate(data, start=1):
                cell = ws.cell(row=row_idx, column=col_idx, value=value)
                cell.border = border
        
        column_widths = [5, 10, 12, 40, 25, 15, 20, 20, 15, 20, 20, 12, 30]
        for idx, width in enumerate(column_widths, start=1):
            col_letter = chr(64 + idx) if idx <= 26 else chr(64 + (idx // 26)) + chr(64 + (idx % 26))
            ws.column_dimensions[col_letter].width = width
            
    elif report_type == 'summary':
        ws = wb.create_sheet("Сводный отчет")
        
        ws['A1'] = 'СВОДНЫЙ ОТЧЕТ ПО ОБРАЩЕНИЯМ'
        ws['A1'].font = Font(bold=True, size=14)
        ws.merge_cells('A1:D1')
        
        headers = ['Отдел', 'Всего', 'В работе', 'Завершено']
        for col_idx, header in enumerate(headers, start=1):
            cell = ws.cell(row=3, column=col_idx, value=header)
            cell.font = header_font
            cell.fill = header_fill
            cell.alignment = header_alignment
            cell.border = border
        
        departments = Department.objects.all()
        row_idx = 4
        for dept in departments:
            dept_appeals = appeals.filter(department=dept)
            total = dept_appeals.count()
            in_progress = dept_appeals.filter(status__name='В работе').count()
            completed = dept_appeals.filter(status__name='Завершенное').count()
            
            ws.cell(row=row_idx, column=1, value=dept.name).border = border
            ws.cell(row=row_idx, column=2, value=total).border = border
            ws.cell(row=row_idx, column=3, value=in_progress).border = border
            ws.cell(row=row_idx, column=4, value=completed).border = border
            row_idx += 1
        
        ws.cell(row=row_idx, column=1, value='ИТОГО:').font = Font(bold=True)
        ws.cell(row=row_idx, column=2, value=appeals.count()).font = Font(bold=True)
        
        ws.column_dimensions['A'].width = 30
        ws.column_dimensions['B'].width = 15
        ws.column_dimensions['C'].width = 15
        ws.column_dimensions['D'].width = 15
        
    else:
        ws = wb.create_sheet("Статистический отчет")
        
        ws['A1'] = 'СТАТИСТИЧЕСКИЙ ОТЧЕТ'
        ws['A1'].font = Font(bold=True, size=14)
        ws.merge_cells('A1:B1')
        
        stats_data = [
            ['Всего обращений', appeals.count()],
            ['Новые', appeals.filter(status__name='Новое').count()],
            ['В работе', appeals.filter(status__name='В работе').count()],
            ['Завершенные', appeals.filter(status__name='Завершенное').count()],
            ['Просроченные', appeals.filter(status__name='Просроченное').count()],
        ]
        
        for row_idx, row_data in enumerate(stats_data, start=3):
            for col_idx, value in enumerate(row_data, start=1):
                cell = ws.cell(row=row_idx, column=col_idx, value=value)
                cell.border = border
                if col_idx == 1:
                    cell.font = Font(bold=True)
        
        ws.column_dimensions['A'].width = 25
        ws.column_dimensions['B'].width = 15
    
    response = HttpResponse(content_type='application/vnd.openxmlformats-officedocument.spreadsheetml.sheet')
    response['Content-Disposition'] = 'attachment; filename="report.xlsx"'
    wb.save(response)
    
    return response


def generate_pdf_report(appeals, report_type, request):
    try:
        from reportlab.lib import colors
        from reportlab.lib.pagesizes import landscape, A4
        from reportlab.platypus import SimpleDocTemplate, Table, TableStyle, Paragraph, Spacer
        from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
        from reportlab.lib.units import mm
    except ImportError:
        return HttpResponse('Библиотека reportlab не установлена. Выполните: pip install reportlab', status=500)
    
    buffer = io.BytesIO()
    doc = SimpleDocTemplate(buffer, pagesize=landscape(A4), topMargin=20*mm, bottomMargin=20*mm)
    
    styles = getSampleStyleSheet()
    title_style = ParagraphStyle('CustomTitle', parent=styles['Heading1'], fontSize=16, alignment=1, spaceAfter=20)
    heading_style = ParagraphStyle('CustomHeading', parent=styles['Heading2'], fontSize=12, 
                                    textColor=colors.HexColor('#1a3a5c'), spaceAfter=10)
    
    elements = []
    elements.append(Paragraph("Отчет по обращениям", title_style))
    elements.append(Spacer(1, 10*mm))
    
    if report_type == 'detailed':
        elements.append(Paragraph("Детальный отчет", heading_style))
        
        data = [['№', 'Номер', 'Дата', 'Тема', 'Заявитель', 'Статус']]
        for idx, appeal in enumerate(appeals[:50], start=1):
            data.append([
                str(idx),
                str(appeal.appeal_id),
                appeal.created_date.strftime('%d.%m.%Y') if appeal.created_date else '-',
                appeal.topic[:50] if appeal.topic else '-',
                f"{appeal.applicant_last_name} {appeal.applicant_first_name}",
                appeal.status.name if appeal.status else '-',
            ])
        
        table = Table(data, colWidths=[20*mm, 25*mm, 30*mm, 60*mm, 40*mm, 30*mm])
        table.setStyle(TableStyle([
            ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#1a3a5c')),
            ('TEXTCOLOR', (0, 0), (-1, 0), colors.white),
            ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
            ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
            ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
            ('FONTSIZE', (0, 1), (-1, -1), 8),
        ]))
        elements.append(table)
        
    elif report_type == 'summary':
        elements.append(Paragraph("Сводный отчет", heading_style))
        
        data = [['Отдел', 'Всего', 'В работе', 'Завершено']]
        departments = Department.objects.all()
        for dept in departments:
            dept_appeals = appeals.filter(department=dept)
            data.append([
                dept.name,
                str(dept_appeals.count()),
                str(dept_appeals.filter(status__name='В работе').count()),
                str(dept_appeals.filter(status__name='Завершенное').count()),
            ])
        data.append(['ИТОГО:', str(appeals.count()), '', ''])
        
        table = Table(data, colWidths=[60*mm, 40*mm, 40*mm, 40*mm])
        table.setStyle(TableStyle([
            ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#1a3a5c')),
            ('TEXTCOLOR', (0, 0), (-1, 0), colors.white),
            ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
            ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
            ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
        ]))
        elements.append(table)
        
    else:
        elements.append(Paragraph("Статистический отчет", heading_style))
        
        data = [
            ['Показатель', 'Количество'],
            ['Всего обращений', str(appeals.count())],
            ['Новые', str(appeals.filter(status__name='Новое').count())],
            ['В работе', str(appeals.filter(status__name='В работе').count())],
            ['Завершенные', str(appeals.filter(status__name='Завершенное').count())],
            ['Просроченные', str(appeals.filter(status__name='Просроченное').count())],
        ]
        
        table = Table(data, colWidths=[80*mm, 40*mm])
        table.setStyle(TableStyle([
            ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#1a3a5c')),
            ('TEXTCOLOR', (0, 0), (-1, 0), colors.white),
            ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
            ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
            ('GRID', (0, 0), (-1, -1), 0.5, colors.grey),
        ]))
        elements.append(table)
    
    doc.build(elements)
    pdf = buffer.getvalue()
    buffer.close()
    
    response = HttpResponse(content_type='application/pdf')
    response['Content-Disposition'] = 'attachment; filename="report.pdf"'
    response.write(pdf)
    
    return response

# Валидаторы
phone_validator = RegexValidator(
    regex=r'^\+?[0-9\s\-\(\)]{10,20}$',
    message='Телефон может содержать только цифры, +, -, пробелы и скобки. Минимум 10 цифр.'
)

building_validator = RegexValidator(
    regex=r'^[a-zA-Z0-9\s\/\-]+$',
    message='Номер дома может содержать только цифры, буквы, / и -'
)

apartment_validator = RegexValidator(
    regex=r'^\d+$',
    message='Номер квартиры может содержать только цифры'
)


class AddressForm(forms.ModelForm):
    country = forms.CharField(
        max_length=100, 
        required=False, 
        widget=forms.TextInput(attrs={'class': 'form-control', 'placeholder': 'Страна'})
    )
    city = forms.CharField(

    const statusFilter = document.getElementById('statusFilter');

    function updateFilters() {
        const url = new URL(window.location.href);
        const selectedStatus = statusFilter.value;
        const selectedDate = dateFilter.value;
        
        if (selectedStatus === 'all') {
            url.searchParams.delete('status');
        } else {
            url.searchParams.set('status', selectedStatus);
        }
        
        if (selectedDate === 'newest') {
            url.searchParams.delete('sort');
        } else {
            url.searchParams.set('sort', selectedDate);
        }
        
        window.location.href = url.toString();
    }
    
    if (statusFilter) {
        statusFilter.addEventListener('change', updateFilters);
    }
    
    if (dateFilter) {
        dateFilter.addEventListener('change', updateFilters);
    }
</script>

        const searchInput = document.getElementById('globalSearch');
        const searchForm = document.getElementById('globalSearchForm');
        const searchResults = document.getElementById('searchResults');
        let searchTimeout = null;
        
        if (searchInput && searchForm) {
            searchInput.addEventListener('input', function() {
                clearTimeout(searchTimeout);
                const query = this.value.trim();
                
                if (query.length < 2) {
                    searchResults.style.display = 'none';
                    return;
                }
                
                searchTimeout = setTimeout(function() {
                    fetch(`/appeals/search-ajax/?q=${encodeURIComponent(query)}`, {
                        headers: {
                            'X-Requested-With': 'XMLHttpRequest'
                        }
                    })
                    .then(response => response.json())
                    .then(data => {
                        if (data.success && data.results.length > 0) {
                            let html = '<div class="search-results-header">🔍 Результаты поиска:</div>';
                            data.results.forEach(item => {
                                html += `
                                    <a href="/appeals/${item.id}/" class="search-result-item">
                                        <div class="result-title">📄 Обращение №${item.appeal_id} - ${item.topic}</div>
                                        <div class="result-details">👤 ${item.applicant} | 📁 ${item.department} | 📊 ${item.status}</div>
                                    </a>
                                `;
                            });
                            searchResults.innerHTML = html;
                            searchResults.style.display = 'block';
                        } else if (data.success && data.results.length === 0) {
                            searchResults.innerHTML = '<div class="search-no-results">❌ Ничего не найдено</div>';
                            searchResults.style.display = 'block';
                        }
                    })
                    .catch(error => console.error('Search error:', error));
                }, 300);
            });
            
            searchInput.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    searchForm.submit();
                }
            });
            
            document.addEventListener('click', function(e) {
                if (!searchInput.contains(e.target) && !searchResults.contains(e.target)) {
                    searchResults.style.display = 'none';
                }
            });
        }
